CREATE TABLE IF NOT EXISTS CAFEPROJECT_USER_TB (
    USER_ID VARCHAR(255) NOT NULL,
    USER_NAME VARCHAR(255) NOT NULL,
    USER_NICKNAME VARCHAR(255) NOT NULL,
    USER_PW VARCHAR(255) NOT NULL,
    USER_IMAGE LONGBLOB NULL,
    USER_IMAGE_TYPE VARCHAR(255) NULL,
    USER_LV INT NOT NULL DEFAULT 1,
    CRT_DT DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CRT_USER VARCHAR(255) NOT NULL,
    UDT_DT DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UDT_USER VARCHAR(255) NOT NULL,
    PRIMARY KEY (USER_ID),
    UNIQUE KEY UK_USER_NICKNAME (USER_NICKNAME)
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8mb4
  COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS CAFEPROJECT_BOARD_TB (
    BOARD_NUMBER BIGINT NOT NULL AUTO_INCREMENT,
    BOARD_CATEGORY INT NOT NULL,
    BOARD_TITLE VARCHAR(255) NOT NULL,
    BOARD_WRITE VARCHAR(1000) NOT NULL,
    USER_ID VARCHAR(255) NOT NULL,
    CRT_DT DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CRT_USER VARCHAR(255) NOT NULL,
    UDT_DT DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UDT_USER VARCHAR(255) NOT NULL,
    PRIMARY KEY (BOARD_NUMBER),
    KEY IDX_BOARD_USER_ID (USER_ID),
    KEY IDX_BOARD_CATEGORY (BOARD_CATEGORY),
    CONSTRAINT FK_BOARD_USER
        FOREIGN KEY (USER_ID) REFERENCES CAFEPROJECT_USER_TB (USER_ID)
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8mb4
  COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS CAFEPROJECT_COMMENT_TB (
    COMMENT_ID BIGINT NOT NULL AUTO_INCREMENT,
    BOARD_NUMBER BIGINT NOT NULL,
    USER_ID VARCHAR(255) NOT NULL,
    PARENT_COMMENT_ID BIGINT NULL,
    COMMENT_CONTENT VARCHAR(1000) NOT NULL,
    DELETED_YN CHAR(1) NOT NULL DEFAULT 'N',
    CRT_DT DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CRT_USER VARCHAR(255) NOT NULL,
    UDT_DT DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UDT_USER VARCHAR(255) NOT NULL,
    PRIMARY KEY (COMMENT_ID),
    KEY IDX_COMMENT_BOARD (BOARD_NUMBER),
    KEY IDX_COMMENT_USER (USER_ID),
    KEY IDX_COMMENT_PARENT (PARENT_COMMENT_ID),
    KEY IDX_COMMENT_BOARD_PARENT (BOARD_NUMBER, PARENT_COMMENT_ID),
    CONSTRAINT FK_COMMENT_BOARD
        FOREIGN KEY (BOARD_NUMBER) REFERENCES CAFEPROJECT_BOARD_TB (BOARD_NUMBER),
    CONSTRAINT FK_COMMENT_USER
        FOREIGN KEY (USER_ID) REFERENCES CAFEPROJECT_USER_TB (USER_ID),
    CONSTRAINT FK_COMMENT_PARENT
        FOREIGN KEY (PARENT_COMMENT_ID) REFERENCES CAFEPROJECT_COMMENT_TB (COMMENT_ID)
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8mb4
  COLLATE=utf8mb4_unicode_ci;
